(defun rust-save-compile-and-run-this ()
  (interactive)
  (save-buffer)
  (compile
   (format "rustc %s && %s"
       (buffer-file-name)
       (file-name-sans-extension (buffer-file-name)))))

(defun rust-save-compile-and-run ()
  (interactive)
  (save-buffer)

  (if (locate-dominating-file (buffer-file-name) "Cargo.toml")
      (compile "cargo run")
    (rust-save-compile-and-run-this)))
